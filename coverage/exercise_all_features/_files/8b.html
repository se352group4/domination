493            return "\\Q" + s + "\\E";
494 
495        StringBuilder sb = new StringBuilder(s.length() * 2);
496        sb.append("\\Q");
497        slashEIndex = 0;
498        int current = 0;

500            sb.append(s.substring(current, slashEIndex));
501            current = slashEIndex + 2;
502            sb.append("\\E\\\\E\\Q");

505        sb.append("\\E");
506        return sb.toString();
